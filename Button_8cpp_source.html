188void Button::tick() {
189 unsigned long now = millis();
190
197 unsigned long timeDelta = now - m_buttonPressedTime;
198
199 switch (m_state) {
200 case State::BUTTON_NOT_PRESSED: {
201 if (isButtonPressed()) {
202 m_state = State::BUTTON_PRESSED;
203 m_buttonPressedTime = now;
204 }
205 break;
206 }
207 case State::BUTTON_PRESSED: {
208 if (isButtonPressed()) {
209 if (timeDelta > m_debounceTicks && !m_buttonPressNotified) {
210 m_buttonPressNotified = true;
211 notifyOnButtonPress();
212 }
213
214 if (timeDelta > m_longPressTicks && !m_isLongButtonPress) {
215 m_isLongButtonPress = true;
216 notifyOnLongPressStart();
217 }
218 } else {
219 if (timeDelta <= m_debounceTicks) {
220 m_state = State::BUTTON_NOT_PRESSED;
221 } else {
222 m_state = State::BUTTON_RELEASED;
223 m_buttonPressNotified = false;
224 notifyOnButtonRelease();
225 }
226 m_buttonReleasedTime = now;
227 }
228 break;
229 }
230 case State::BUTTON_RELEASED: {
231 if (isButtonPressed() && (now - m_buttonReleasedTime) > m_debounceTicks) {
232 m_buttonPressedTime = now;
233 m_state = State::BUTTON_DOUBLE_CLICKED;
234 } else if (timeDelta > m_longPressTicks) {
235 m_isLongButtonPress = false;
236 m_state = State::BUTTON_NOT_PRESSED;
237 notifyOnLongPressEnd();
238 } else if (timeDelta > m_clickTicks) {
239 m_state = State::BUTTON_NOT_PRESSED;
240 notifyOnClick();
241 }
242 break;
243 }
244 case State::BUTTON_DOUBLE_CLICKED: {
245 if (!isButtonPressed() && timeDelta > m_debounceTicks) {
246 m_state = State::BUTTON_NOT_PRESSED;
247 notifyOnDoubleClick();
248 }
249 break;
250 }
251 }
252}
