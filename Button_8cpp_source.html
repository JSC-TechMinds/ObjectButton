190void Button::tick() {
191 unsigned long now = millis();
192
199 unsigned long timeDelta = now - m_buttonPressedTime;
200
201 switch (m_state) {
202 case State::BUTTON_NOT_PRESSED: {
203 if (isButtonPressed()) {
204 m_state = State::BUTTON_PRESSED;
205 m_buttonPressedTime = now;
206 }
207 break;
208 }
209 case State::BUTTON_PRESSED: {
210 if (isButtonPressed()) {
211 if (timeDelta > m_debounceTicks && !m_buttonPressNotified) {
212 m_buttonPressNotified = true;
213 notifyOnButtonPress();
214 }
215
216 if (timeDelta > m_longPressTicks && !m_isLongButtonPress) {
217 m_isLongButtonPress = true;
218 notifyOnLongPressStart();
219 }
220 } else {
221 if (timeDelta <= m_debounceTicks) {
222 m_state = State::BUTTON_NOT_PRESSED;
223 } else {
224 m_state = State::BUTTON_RELEASED;
225 m_buttonPressNotified = false;
226 notifyOnButtonRelease();
227 }
228 m_buttonReleasedTime = now;
229 }
230 break;
231 }
232 case State::BUTTON_RELEASED: {
233 if (isButtonPressed() && (now - m_buttonReleasedTime) > m_debounceTicks) {
234 m_buttonPressedTime = now;
235 m_state = State::BUTTON_DOUBLE_CLICKED;
236 } else if (timeDelta > m_longPressTicks) {
237 m_isLongButtonPress = false;
238 m_state = State::BUTTON_NOT_PRESSED;
239 notifyOnLongPressEnd();
240 } else if (timeDelta > m_clickTicks) {
241 m_state = State::BUTTON_NOT_PRESSED;
242 notifyOnClick();
243 }
244 break;
245 }
246 case State::BUTTON_DOUBLE_CLICKED: {
247 if (!isButtonPressed() && timeDelta > m_debounceTicks) {
248 m_state = State::BUTTON_NOT_PRESSED;
249 notifyOnDoubleClick();
250 }
251 break;
252 }
253 }
254}
